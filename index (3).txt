    function updateClock() {
      const now = new Date();

      const h = String(hawaii.getHours()).padStart(2, '0');
      const m = String(hawaii.getMinutes()).padStart(2, '0');
      const s = String(hawaii.getSeconds()).padStart(2, '0');
      document.getElementById('clock').textContent = `${h}:${m}:${s}`;

      const div = document.createElement('div');

      document.getElementById('foro-nombre').value = '';
      document.getElementById('foro-texto').value = '';
